211 static inline neu_msg_t *neu_msg_new(neu_reqresp_type_e t, void *ctx,
212  void *data)
213 {
214  //获取消息对象字节大小
215  size_t data_size = neu_reqresp_size(t);
216 
217  //存储消息体的大小，初始化为 0
218  size_t body_size = 0;
219 
220  //根据消息类型 t 来计算消息体（body）的大小。
221  switch (t) {
222  case NEU_REQ_CHECK_SCHEMA:
223  body_size = neu_reqresp_size(NEU_RESP_CHECK_SCHEMA);
224  break;
225  case NEU_REQ_GET_PLUGIN:
226  body_size = neu_reqresp_size(NEU_RESP_GET_PLUGIN);
227  break;
228  case NEU_REQ_UPDATE_GROUP:
229  case NEU_REQ_UPDATE_DRIVER_GROUP:
230  body_size = neu_reqresp_size(NEU_RESP_UPDATE_DRIVER_GROUP);
231  break;
232  case NEU_REQ_UPDATE_NODE:
233  case NEU_REQ_NODE_RENAME:
234  body_size = neu_reqresp_size(NEU_RESP_NODE_RENAME);
235  break;
236  case NEU_REQ_DEL_NODE:
237  body_size = neu_reqresp_size(NEU_RESP_NODE_UNINIT);
238  break;
239  case NEU_REQ_GET_NODE_SETTING:
240  body_size = neu_reqresp_size(NEU_RESP_GET_NODE_SETTING);
241  break;
242  case NEU_REQ_GET_NODES_STATE:
243  body_size = neu_reqresp_size(NEU_RESP_GET_NODES_STATE);
244  break;
245  default:
246  body_size = data_size;
247  }
248 
249  //计算消息对象的总大小: sizeof(neu_msg_t)只计算柔性输入之前的大小即neu_msg_s
250  size_t total = sizeof(neu_msg_t) + body_size;
251 
252  neu_msg_t *msg = calloc(1, total);
253  if (msg) {
254  msg->head.type = t; // 设置为传入的消息类型
255  msg->head.len = total; // 设置为消息对象的总大小
256  msg->head.ctx = ctx; // 设置为传入的上下文指针
257  if (data) { // 传入的数据指针 data 不为空
258  memcpy(msg->body, data, data_size);
259  }
260  }
261  return msg;
262 }
