538 static inline int neu_node_metrics_add(neu_node_metrics_t *node_metrics,
539  const char *group_name, const char *name,
540  const char *help, neu_metric_type_e type,
541  uint64_t init)
542 {
543  int rv = 0;
544 
545  // 尝试注册度量项，如果失败则直接返回错误码
546  if (0 > neu_metrics_register_entry(name, help, type)) {
547  return -1;
548  }
549 
550  pthread_mutex_lock(&node_metrics->lock);
551  if (NULL == group_name) {
552  // 如果没有提供组名，则直接将度量项添加到节点度量中
553  rv = neu_metric_entries_add(&node_metrics->entries, name, help, type,
554  init);
555  } else {
556  neu_group_metrics_t *group_metrics = NULL;
557 
558  // 查找是否存在指定名称的组
559  HASH_FIND_STR(node_metrics->group_metrics, group_name, group_metrics);
560  if (NULL != group_metrics) {
561  // 如果组存在，则将度量项添加到该组中
562  rv = neu_metric_entries_add(&group_metrics->entries, name, help,
563  type, init);
564  } else {
565  // 如果组不存在，则创建新组并将度量项添加到新组中
566  group_metrics =
567  (neu_group_metrics_t *) calloc(1, sizeof(*group_metrics));
568  if (NULL != group_metrics &&
569  NULL != (group_metrics->name = strdup(group_name))) {
570  HASH_ADD_STR(node_metrics->group_metrics, name, group_metrics);
571  rv = neu_metric_entries_add(&group_metrics->entries, name, help,
572  type, init);
573  } else {
574  free(group_metrics);
575  rv = -1;
576  }
577  }
578  }
579  pthread_mutex_unlock(&node_metrics->lock);
580 
581  // 如果添加度量项失败，则取消注册度量项
582  if (0 != rv) {
583  neu_metrics_unregister_entry(name);
584  }
585  return rv;
586 }

601 static inline int neu_node_metrics_update(neu_node_metrics_t *node_metrics,
602  const char * group,
603  const char *metric_name, uint64_t n)
604 {
605  neu_metric_entry_t *entry = NULL;
606 
607  pthread_mutex_lock(&node_metrics->lock);
608 
609  // 根据组名查找指标项
610  if (NULL == group) {
611  HASH_FIND_STR(node_metrics->entries, metric_name, entry); // 表中查找指标名称的项
612  } else if (NULL != node_metrics->group_metrics) {
613  neu_group_metrics_t *g = NULL;
614  HASH_FIND_STR(node_metrics->group_metrics, group, g); //查找组名为 group 的组
615  if (NULL != g) {
616  HASH_FIND_STR(g->entries, metric_name, entry);
617  }
618  }
619 
620  if (NULL == entry) {
621  pthread_mutex_unlock(&node_metrics->lock);
622  return -1;
623  }
624 
625  //判断度量的类型
626  if (neu_metric_type_is_counter(entry->type)) {
627  entry->value += n;
628  } else if (neu_metric_type_is_rolling_counter(entry->type)) {
629  entry->value =
630  neu_rolling_counter_inc(entry->rcnt, global_timestamp, n);
631  } else {
632  entry->value = n;
633  }
634  pthread_mutex_unlock(&node_metrics->lock);
635 
636  return 0;
637 }

639 static inline void neu_node_metrics_reset(neu_node_metrics_t *node_metrics)
640 {
641  neu_metric_entry_t *entry = NULL;
642 
643  pthread_mutex_lock(&node_metrics->lock);
644  HASH_LOOP(hh, node_metrics->entries, entry)
645  {
646  if (!neu_metric_type_no_reset(entry->type)) {
647  entry->value = entry->init;
648  if (neu_metric_type_is_rolling_counter(entry->type)) {
649  neu_rolling_counter_reset(entry->rcnt);
650  }
651  }
652  }
653 
654  neu_group_metrics_t *g = NULL;
655  HASH_LOOP(hh, node_metrics->group_metrics, g)
656  {
657  HASH_LOOP(hh, g->entries, entry)
658  {
659  if (!neu_metric_type_no_reset(entry->type)) {
660  entry->value = entry->init;
661  if (neu_metric_type_is_rolling_counter(entry->type)) {
662  neu_rolling_counter_reset(entry->rcnt);
663  }
664  }
665  }
666  }
667  pthread_mutex_unlock(&node_metrics->lock);
668 }
